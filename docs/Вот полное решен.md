## **1. Хранилище аутентификации**

```javascript
// src/lib/stores/auth.js
import { writable, derived } from 'svelte/store';
import { browser } from '$app/environment';

// Функция парсинга JWT
function parseJwt(token) {
  try {
    const base64Url = token.split('.')[1];
    const base64 = base64Url.replace(/-/g, '+').replace(/_/g, '/');
    const jsonPayload = decodeURIComponent(
      atob(base64)
        .split('')
        .map(c => '%' + ('00' + c.charCodeAt(0).toString(16)).slice(-2))
        .join('')
    );
    return JSON.parse(jsonPayload);
  } catch {
    return null;
  }
}

// Проверка срока действия токена
function isTokenExpired(token) {
  const payload = parseJwt(token);
  if (!payload || !payload.exp) return true;
  return payload.exp * 1000 < Date.now();
}

// Создаем хранилище
function createAuthStore() {
  const { subscribe, set, update } = writable({
    user: null,
    token: null,
    isAuthenticated: false,
    isLoading: true
  });

  // Инициализация
  if (browser) {
    const token = localStorage.getItem('auth_token');
    if (token && !isTokenExpired(token)) {
      const payload = parseJwt(token);
      set({
        user: {
          id: payload.sub || payload.nameid,
          username: payload.unique_name || payload.email,
          email: payload.email,
          roles: payload.role 
            ? (Array.isArray(payload.role) ? payload.role : [payload.role])
            : []
        },
        token,
        isAuthenticated: true,
        isLoading: false
      });
    } else {
      localStorage.removeItem('auth_token');
      set({ user: null, token: null, isAuthenticated: false, isLoading: false });
    }
  }

  return {
    subscribe,
    login: (token) => {
      localStorage.setItem('auth_token', token);
      const payload = parseJwt(token);
      set({
        user: {
          id: payload.sub || payload.nameid,
          username: payload.unique_name || payload.email,
          email: payload.email,
          roles: payload.role 
            ? (Array.isArray(payload.role) ? payload.role : [payload.role])
            : []
        },
        token,
        isAuthenticated: true,
        isLoading: false
      });
    },
    logout: () => {
      localStorage.removeItem('auth_token');
      set({ user: null, token: null, isAuthenticated: false, isLoading: false });
    },
    refresh: () => {
      if (browser) {
        const token = localStorage.getItem('auth_token');
        if (token && !isTokenExpired(token)) {
          const payload = parseJwt(token);
          set({
            user: {
              id: payload.sub || payload.nameid,
              username: payload.unique_name || payload.email,
              email: payload.email,
              roles: payload.role 
                ? (Array.isArray(payload.role) ? payload.role : [payload.role])
                : []
            },
            token,
            isAuthenticated: true,
            isLoading: false
          });
        } else {
          localStorage.removeItem('auth_token');
          set({ user: null, token: null, isAuthenticated: false, isLoading: false });
        }
      }
    }
  };
}

export const auth = createAuthStore();

// Деривед-сторе для проверки ролей
export const hasRole = derived(auth, ($auth) => 
  (...roles) => {
    if (!$auth.user || !$auth.user.roles) return false;
    return roles.some(role => $auth.user.roles.includes(role));
  }
);
```

## **2. Сервис для работы с API**

```javascript
// src/lib/services/api.js
import { auth } from '$lib/stores/auth';

const API_URL = import.meta.env.VITE_API_URL || 'https://your-api.com';

async function fetchWithAuth(url, options = {}) {
  const { token } = auth;

  const headers = {
    'Content-Type': 'application/json',
    ...options.headers
  };

  if (token) {
    headers['Authorization'] = `Bearer ${token}`;
  }

  const response = await fetch(`${API_URL}${url}`, {
    ...options,
    headers
  });

  // Если 401 - разлогиниваем
  if (response.status === 401) {
    auth.logout();
    throw new Error('Unauthorized');
  }

  if (!response.ok) {
    throw new Error(`HTTP ${response.status}: ${await response.text()}`);
  }

  return response;
}

export const api = {
  async login(credentials) {
    const response = await fetch(`${API_URL}/api/auth/login`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(credentials)
    });

    if (!response.ok) throw new Error('Login failed');

    const data = await response.json();
    auth.login(data.token);
    return data;
  },

  async register(userData) {
    const response = await fetch(`${API_URL}/api/auth/register`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(userData)
    });

    if (!response.ok) throw new Error('Registration failed');
    return await response.json();
  },

  async getUserProfile() {
    const response = await fetchWithAuth('/api/auth/profile');
    return await response.json();
  },

  // Общие методы
  get: (url) => fetchWithAuth(url, { method: 'GET' }),
  post: (url, data) => fetchWithAuth(url, { 
    method: 'POST', 
    body: JSON.stringify(data) 
  }),
  put: (url, data) => fetchWithAuth(url, { 
    method: 'PUT', 
    body: JSON.stringify(data) 
  }),
  delete: (url) => fetchWithAuth(url, { method: 'DELETE' })
};
```

## **3. Защищенный Layout (аналог [Authorize] в Blazor)**

```svelte
<!-- src/routes/+layout.svelte -->
<script>
  import { auth } from '$lib/stores/auth';
  import { onMount } from 'svelte';
  import { page } from '$app/stores';
  import { goto } from '$app/navigation';

  let unsubscribe;

  onMount(() => {
    // Подписываемся на изменения аутентификации
    unsubscribe = auth.subscribe(($auth) => {
      // Если не авторизован и не на странице логина - редирект
      if (!$auth.isLoading && !$auth.isAuthenticated) {
        const currentPath = window.location.pathname;
        const publicRoutes = ['/login', '/register', '/forgot-password'];

        if (!publicRoutes.includes(currentPath)) {
          goto('/login');
        }
      }

      // Если авторизован и на странице логина - редирект на главную
      if ($auth.isAuthenticated && window.location.pathname === '/login') {
        goto('/');
      }
    });
  });
</script>

<slot />

<style global>
  /* Глобальные стили */
</style>
```

## **4. Страница логина**

```svelte
<!-- src/routes/login/+page.svelte -->
<script>
  import { onMount } from 'svelte';
  import { auth, api } from '$lib/services/api';
  import { goto } from '$app/navigation';

  let username = '';
  let password = '';
  let error = '';
  let isLoading = false;

  // Если уже авторизован - редирект
  onMount(() => {
    const unsubscribe = auth.subscribe(($auth) => {
      if ($auth.isAuthenticated) {
        goto('/');
      }
    });
    return unsubscribe;
  });

  async function handleLogin(e) {
    e.preventDefault();
    error = '';
    isLoading = true;

    try {
      await api.login({ username, password });
      goto('/');
    } catch (err) {
      error = err.message || 'Ошибка входа';
    } finally {
      isLoading = false;
    }
  }
</script>

<div class="login-page">
  <form on:submit={handleLogin}>
    <input 
      bind:value={username}
      placeholder="Имя пользователя" 
      required
    />

    <input 
      type="password"
      bind:value={password}
      placeholder="Пароль" 
      required
    />

    {#if error}
      <div class="error">{error}</div>
    {/if}

    <button type="submit" disabled={isLoading}>
      {isLoading ? 'Вход...' : 'Войти'}
    </button>
  </form>
</div>

<style>
  .login-page {
    /* стили формы */
  }

  .error {
    color: red;
  }
</style>
```

## **5. Компонент для защиты отдельных элементов**

```svelte
<!-- src/lib/components/AuthGuard.svelte -->
<script>
  import { auth } from '$lib/stores/auth';
  import { goto } from '$app/navigation';

  export let fallback = '/login';
  export let roles = [];

  // Автоматический редирект если не авторизован
  $: if (auth.isLoading === false && !auth.isAuthenticated) {
    goto(fallback);
  }

  // Проверка ролей
  $: if (roles.length > 0 && auth.user) {
    const hasRequiredRole = roles.some(role => 
      auth.user.roles.includes(role)
    );
    if (!hasRequiredRole) {
      goto('/unauthorized');
    }
  }
</script>

{#if auth.isAuthenticated}
  <slot />
{:else if auth.isLoading}
  <div>Загрузка...</div>
{/if}
```

## **8. ASP.NET Core конфигурация (C#)**

```csharp
// Program.cs
using Microsoft.AspNetCore.Authentication.JwtBearer;
using Microsoft.IdentityModel.Tokens;
using System.Text;

var builder = WebApplication.CreateBuilder(args);

// Настройка JWT
builder.Services.AddAuthentication(options =>
{
    options.DefaultAuthenticateScheme = JwtBearerDefaults.AuthenticationScheme;
    options.DefaultChallengeScheme = JwtBearerDefaults.AuthenticationScheme;
})
.AddJwtBearer(options =>
{
    options.TokenValidationParameters = new TokenValidationParameters
    {
        ValidateIssuer = true,
        ValidateAudience = true,
        ValidateLifetime = true,
        ValidateIssuerSigningKey = true,
        ValidIssuer = builder.Configuration["Jwt:Issuer"],
        ValidAudience = builder.Configuration["Jwt:Audience"],
        IssuerSigningKey = new SymmetricSecurityKey(
            Encoding.UTF8.GetBytes(builder.Configuration["Jwt:Key"]))
    };

    // Для SvelteKit CORS
    options.Events = new JwtBearerEvents
    {
        OnMessageReceived = context =>
        {
            var token = context.Request.Headers["Authorization"].FirstOrDefault()?.Split(" ").Last();
            if (!string.IsNullOrEmpty(token))
            {
                context.Token = token;
            }
            return Task.CompletedTask;
        }
    };
});

// Настройка CORS для SvelteKit
builder.Services.AddCors(options =>
{
    options.AddPolicy("SvelteKit", policy =>
    {
        policy.WithOrigins("http://localhost:5173") // dev
              .AllowAnyHeader()
              .AllowAnyMethod()
              .AllowCredentials();
    });
});

var app = builder.Build();

app.UseCors("SvelteKit");
app.UseAuthentication();
app.UseAuthorization();

// Контроллер авторизации
app.MapPost("/api/auth/login", async (LoginRequest request) =>
{
    // Ваша логика проверки пользователя
    var user = await userService.Authenticate(request.Username, request.Password);

    if (user == null)
        return Results.Unauthorized();

    var token = GenerateJwtToken(user);

    return Results.Ok(new { 
        token,

            user.Id, 
            user.Username, 
            user.Email,
            user.Roles 
        } 
    });
});

app.Run();
```

## **Как это работает:**

1. **При загрузке приложения** проверяется токен в localStorage
2. **Все маршруты защищены** через корневой layout
3. **JWT токен** автоматически добавляется в заголовки запросов
4. **При 401 ошибке** автоматический выход
5. **Роли** проверяются на уровне компонентов

Этот подход максимально похож на `[Authorize]` в Blazor, но реализован для SvelteKit CSR.
